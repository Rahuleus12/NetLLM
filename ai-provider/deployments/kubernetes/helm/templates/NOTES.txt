================================================================================
                           ACCESSING AI PROVIDER
================================================================================

{{- if .Values.ingress.enabled }}

1. Get the application URL:

{{- range .Values.ingress.hosts }}
  export AI_PROVIDER_URL=https://{{ .host }}
{{- end }}

2. Verify the ingress is ready:
  kubectl get ingress -n {{ .Release.Namespace }} {{ .Release.Name }}

3. Access the API:
  curl $AI_PROVIDER_URL/health

{{- else if contains "NodePort" .Values.service.type }}

1. Get the application URL:

  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "ai-provider.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  export AI_PROVIDER_URL=http://$NODE_IP:$NODE_PORT

2. Access the API:
  curl $AI_PROVIDER_URL/health

{{- else if contains "LoadBalancer" .Values.service.type }}

1. Get the application URL:

  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status by running:
        kubectl get svc -n {{ .Release.Namespace }} -w {{ include "ai-provider.fullname" . }}

  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "ai-provider.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  export AI_PROVIDER_URL=http://$SERVICE_IP:{{ .Values.service.port }}

2. Access the API:
  curl $AI_PROVIDER_URL/health

{{- else if contains "ClusterIP" .Values.service.type }}

1. Get the application URL:

  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "ai-provider.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME {{ .Values.service.port }}:$CONTAINER_PORT &

  export AI_PROVIDER_URL=http://127.0.0.1:{{ .Values.service.port }}

2. Access the API:
  curl $AI_PROVIDER_URL/health

{{- end }}

================================================================================
                           QUICK START
================================================================================

1. Login to the API:

  export ADMIN_USER=$(kubectl get secret -n {{ .Release.Namespace }} {{ .Release.Name }}-admin-credentials -o jsonpath='{.data.ADMIN_USERNAME}' | base64 -d)
  export ADMIN_PASS=$(kubectl get secret -n {{ .Release.Namespace }} {{ .Release.Name }}-admin-credentials -o jsonpath='{.data.ADMIN_PASSWORD}' | base64 -d)

  curl -X POST $AI_PROVIDER_URL/api/v1/auth/login \
    -H "Content-Type: application/json" \
    -d "{\"username\":\"$ADMIN_USER\",\"password\":\"$ADMIN_PASS\"}"

2. List available models:

  curl -X GET $AI_PROVIDER_URL/api/v1/models \
    -H "Authorization: Bearer <your-jwt-token>"

3. Download your first model:

  curl -X POST $AI_PROVIDER_URL/api/v1/models \
    -H "Authorization: Bearer <your-jwt-token>" \
    -H "Content-Type: application/json" \
    -d '{
      "name": "llama-2-7b",
      "source_url": "https://huggingface.co/meta-llama/Llama-2-7b-chat-hf",
      "version": "1.0.0"
    }'

4. Run inference:

  curl -X POST $AI_PROVIDER_URL/api/v1/inference \
    -H "Authorization: Bearer <your-jwt-token>" \
    -H "Content-Type: application/json" \
    -d '{
      "model": "llama-2-7b",
      "prompt": "Hello, how are you?",
      "max_tokens": 100
    }'

                           TROUBLESHOOTING
================================================================================

If your pods are not starting, check:

   kubectl describe pods -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "ai-provider.name" . }}

2. Pod logs:
   kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "ai-provider.name" . }} --all-containers

3. Check if persistent volumes are bound:
   kubectl get pvc -n {{ .Release.Namespace }}

4. Check if database is ready:
{{- if .Values.postgresql.enabled }}
   kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/name=postgresql
{{- end }}

5. Check if Redis is ready:
{{- if .Values.redis.enabled }}
   kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/name=redis

⚠️  IMPORTANT: Change default passwords immediately!

1. Update admin password:

     -p='{"data":{"ADMIN_PASSWORD":"'$(echo -n 'NEW_PASSWORD' | base64)'"} }'

{{- if .Values.postgresql.enabled }}

     -p='{"data":{"password":"'$(echo -n 'NEW_PASSWORD' | base64)'"} }'
   kubectl rollout restart deployment/{{ include "ai-provider.fullname" . }} -n {{ .Release.Namespace }}

3. Update JWT secret:
